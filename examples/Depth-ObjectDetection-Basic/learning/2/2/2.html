# depth 데이터에서 이미지의 높이와 너비 가져오기
height, width = depth.shape

# 좌측 끝 (중심 Y축 기준) 가로 100px x 세로 60px 영역 평균 계산
center_y = height // 2
start_y, end_y = center_y - 30, center_y + 30

left_region = depth[start_y:end_y, :100]
left_mean = np.mean(left_region)

# 우측 끝 (중심 Y축 기준) 가로 100px x 세로 60px 영역 평균 계산
right_region = depth[start_y:end_y, -100:]
right_mean = np.mean(right_region)

print(f"좌측 끝 100x60 영역의 평균 거리값: {left_mean:.2f}")
print(f"우측 끝 100x60 영역의 평균 거리값: {right_mean:.2f}")

if(left_mean > right_mean):
    jajucha2.control.set_motor(10,10,3)
elif(left_mean <= right_mean):
    jajucha2.control.set_motor(-10,-10,3)
